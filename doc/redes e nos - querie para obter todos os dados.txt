SELECT * FROM WATERLINES INNER JOIN X_MATERIAL ON WATERLINES.MATERIAL = X_MATERIAL.MATERIALID 
INNER JOIN X_LOCATION ON WATERLINES.LOCATION = X_LOCATION.LOCATIONID
INNER JOIN X_SIDESTREET ON WATERLINES.SIDESTREET = X_SIDESTREET.SIDESTREET_ID
INNER JOIN X_SUPPLIERS ON WATERLINES.SUPPLIER = X_SUPPLIERS.SUPPLIERID
INNER JOIN X_STATE ON WATERLINES.STATE = X_STATE.STATEID
INNER JOIN WATERLINESTYPES ON WATERLINES.ID_TYPE = WATERLINESTYPES.ID_TYPE
INNER JOIN X_MANUFACTURERS ON WATERLINES.MANUFACTURER = X_MANUFACTURERS.MANUFACTURERID
